11 def merge( 

12 self, 

13 table_name: str, 

14 dataframe: pd.DataFrame, 

15 match_columns: List[str] = None, 

16 upsert: bool = False,

19 """Merge a dataframe into an SQL table by updating, inserting, and/or deleting rows using Transact-SQL MERGE. 

20 With upsert=True, an update if exists otherwise insert action is performed without deleting anything. The 

21 delete_requires parameter allows for incremental merging by preventing records from being deleted if at least 

22 one record doesn't match in the column. 

23 

24 Parameters 

25 ---------- 

26 

27 table_name (str) : name of the SQL table 

28 dataframe (pandas.DataFrame): tabular data to merge into SQL table 

29 match_columns (list, default=None) : combination of columns or index to determine matches, if None the SQL primary key is used 

30 upsert (bool, default=False) : delete records if they do not match 

31 delete_requires (list, default=None) : column(s) that need to have a matching row for records to be deleted 

32 

33 Returns 

34 ------- 

35 dataframe (pandas.DataFrame) : input dataframe that may have been altered to conform to SQL 

36 

37 Examples 

38 -------- 

39 #### merge ColumnA and ColumnB values based on the SQL primary key / index of the dataframe 

40 write.merge('SomeTable', dataframe[['ColumnA','ColumnB']]) 

41 

42 #### for incrementally merging from a dataframe, require ColumnC also matches to prevent a record from being deleted 

43 write.merge('SomeTable', dataframe[['ColumnA','ColumnB', 'ColumnC']], delete_requires=['ColumnC']) 

44 

45 #### perform an UPSERT (if exists update, otherwise update) workflow 

46 write.merge('SomeTable', dataframe[['ColumnA']], upsert=True) 

47 

48 """ 

49 # check inputs 

50 if delete_requires is not None and upsert: 

51 raise ValueError("delete_requires can only be specified if upsert==False") 

52 

53 # prevent setwithcopy errors incase a subset of columns from an original dataframe are being updated 

54 dataframe = dataframe.copy() 

55 

56 # create cursor to perform operations 

57 cursor = self._connection.cursor() 

58 

59 # get target table schema, while checking for errors and adjusting data for inserting 

60 if self.include_metadata_timestamps: 

61 additional_columns = ["_time_update", "_time_insert"] 

62 else: 

63 additional_columns = None 

64 schema, dataframe, match_columns, temp_name = self._source_table( 

65 table_name, dataframe, cursor, match_columns, additional_columns 

66 ) 

67 

68 # develop basic merge syntax 

69 statement = """ 

70 DECLARE @SQLStatement AS NVARCHAR(MAX); 

71 DECLARE @TableName SYSNAME = ?; 

72 DECLARE @TableTemp SYSNAME = ?; 

73 {declare} 

74 

75 SET @SQLStatement = 

76 N' MERGE '+QUOTENAME(@TableName)+' AS _target ' 

77 +' USING '+QUOTENAME(@TableTemp)+' AS _source ' 

78 +' ON ('+{match_syntax}+') ' 

79 +' WHEN MATCHED THEN UPDATE SET '+{update_syntax} 

80 +' WHEN NOT MATCHED THEN INSERT ('+{insert_syntax}+')' 

81 +' VALUES ('+{insert_values}+')' 

82 +{delete_syntax}+';' 

83 

84 EXEC sp_executesql 

85 @SQLStatement, 

86 N'@TableName SYSNAME, @TableTemp SYSNAME, {parameters}', 

87 @TableName=@TableName, @TableTemp=@TableTemp, {values}; 

88 """ 

89 

90 # if matched, update all columns in dataframe besides match_columns 

91 update_columns = list(dataframe.columns[~dataframe.columns.isin(match_columns)]) 

92 

93 # if not matched, insert all columns in dataframe 

94 if any(dataframe.index.names): 

95 insert_columns = list(dataframe.index.names) + list(dataframe.columns) 

96 else: 

97 insert_columns = list(dataframe.columns) 

98 

99 # alias columns to prevent direct input into SQL string 

100 alias_match = [str(x) for x in list(range(0, len(match_columns)))] 

101 alias_update = [str(x) for x in list(range(0, len(update_columns)))] 

102 alias_insert = [str(x) for x in list(range(0, len(insert_columns)))] 

103 if delete_requires is None: 

104 alias_conditions = [] 

105 else: 

106 alias_conditions = [str(x) for x in list(range(0, len(delete_requires)))] 

107 

108 # declare SQL variables 

109 declare = ["DECLARE @Match_" + x + " SYSNAME = ?;" for x in alias_match] 

110 declare += ["DECLARE @Update_" + x + " SYSNAME = ?;" for x in alias_update] 

111 declare += ["DECLARE @Insert_" + x + " SYSNAME = ?;" for x in alias_insert] 

112 declare += ["DECLARE @Subset_" + x + " SYSNAME = ?;" for x in alias_conditions] 

113 declare = "\n".join(declare) 

114 

115 # form match on syntax 

116 match_syntax = ["QUOTENAME(@Match_" + x + ")" for x in alias_match] 

117 match_syntax = "+' AND '+".join( 

118 ["'_target.'+" + x + "+'=_source.'+" + x for x in match_syntax] 

119 ) 

120 

121 # form when matched then update syntax 

122 update_syntax = ["QUOTENAME(@Update_" + x + ")" for x in alias_update] 

123 update_syntax = "+','+".join([x + "+'=_source.'+" + x for x in update_syntax]) 

124 if self.include_metadata_timestamps: 

125 update_syntax = "+'_time_update=GETDATE(), '+" + update_syntax 

126 

127 # form when not matched then insert 

128 insert_syntax = "+','+".join( 

129 ["QUOTENAME(@Insert_" + x + ")" for x in alias_insert] 

130 ) 

131 insert_values = "+','+".join( 

132 ["'_source.'+QUOTENAME(@Insert_" + x + ")" for x in alias_insert] 

133 ) 

134 if self.include_metadata_timestamps: 

135 insert_syntax = "+'_time_insert, '+" + insert_syntax 

136 insert_values = "+'GETDATE(), '+" + insert_values 

137 

138 # form when not matched by source then delete condition syntax 

139 if not upsert: 

140 delete_syntax = ( 

141 "' WHEN NOT MATCHED BY SOURCE '+{conditions_syntax}+' THEN DELETE'" 

142 ) 

143 conditions_syntax = [ 

144 "'AND _target.'+QUOTENAME(@Subset_" 

145 + x 

146 + ")+' IN (SELECT '+QUOTENAME(@Subset_" 

147 + x 

148 + ")+' FROM '+QUOTENAME(@TableTemp)+')'" 

149 for x in alias_conditions 

150 ] 

151 conditions_syntax = " + ".join(conditions_syntax) 

152 delete_syntax = delete_syntax.format(conditions_syntax=conditions_syntax) 

153 else: 

154 delete_syntax = "''" 

155 

156 # parameters for sp_executesql 

157 parameters = ["@Match_" + x + " SYSNAME" for x in alias_match] 

158 parameters += ["@Update_" + x + " SYSNAME" for x in alias_update] 

159 parameters += ["@Insert_" + x + " SYSNAME" for x in alias_insert] 

160 parameters += ["@Subset_" + x + " SYSNAME" for x in alias_conditions] 

161 parameters = ", ".join(parameters) 

162 

163 # values for sp_executesql 

164 values = ["@Match_" + x + "=@Match_" + x for x in alias_match] 

165 values += ["@Update_" + x + "=@Update_" + x for x in alias_update] 

166 values += ["@Insert_" + x + "=@Insert_" + x for x in alias_insert] 

167 values += ["@Subset_" + x + "=@Subset_" + x for x in alias_conditions] 

168 values = ", ".join(values) 

169 

170 # set final SQL string 

171 statement = statement.format( 

172 declare=declare, 

173 match_syntax=match_syntax, 

174 update_syntax=update_syntax, 

175 insert_syntax=insert_syntax, 

176 insert_values=insert_values, 

177 delete_syntax=delete_syntax, 

178 parameters=parameters, 

179 values=values, 

180 ) 

181 

182 # perform merge 

183 if delete_requires is None: 

184 args = ( 

185 [table_name, temp_name] 

186 + match_columns 

187 + update_columns 

188 + insert_columns 

189 ) 

190 else: 

191 args = ( 

192 [table_name, temp_name] 

193 + match_columns 

194 + update_columns 

195 + insert_columns 

196 + delete_requires 

197 ) 

198 

199 # execute statement to perform update in target table using source 

200 cursor.execute(statement, args) 

201 temp_name = dynamic.escape(cursor, temp_name) 

202 cursor.execute("DROP TABLE " + temp_name) 

203 cursor.commit() 

204 

205 return dataframe
