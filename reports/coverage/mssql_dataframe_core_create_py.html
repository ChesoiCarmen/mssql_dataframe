67 {pk}

209 def table_from_dataframe( 

210 self, 

211 table_name: str, 

212 dataframe: pd.DataFrame, 

213 primary_key: Literal[None, "sql", "index", "infer"] = None, 

214 insert_dataframe: bool = True, 

215 ) -> pd.DataFrame: 

216 """Create SQL table by inferring SQL create table parameters from the contents of a dataframe. The contents 

217 can be composed of strings/objects only and converted better data types if conversion is possible within pandas. 

218 

219 Parameters 

220 ---------- 

221 table_name (str) : name of table 

222 dataframe (pandas.DataFrame) : data used to create table 

223 primary_key (str, default = None) : method of setting the table's primary key, see below for description of options 

224 insert_dataframe (bool, default=True) : insert the dataframe after creating the table 

225 

226 primary_key = None : do not set a primary key 

227 primary_key = 'sql' : create an SQL managed auto-incrementing INT identity primary key column named '_pk' 

228 primary_key = 'index' : use the index of the dataframe and it's name, or '_index' if the index is not named 

229 primary_key = 'infer' : determine the column in the dataframe that best serves as a primary key and use it's name 

230 

231 Returns 

232 ------- 

233 dataframe (pandas.DataFrame) : data potentially converted from obejcts/strings to better pandas types 

234 

235 Examples 

236 -------- 

237 #### create table without a primary key 

238 df = create.table_from_dataframe('##DFNoPK', pd.DataFrame({"ColumnA": [1]})) 

239 

240 #### create table with the dataframe's index as the primary key 

241 df = create.table_from_dataframe('##DFIndexPK', pd.DataFrame({"ColumnA": [1,2]}, index=['a','z']), primary_key='index') 

242 

243 #### create an SQL identity primary key 

244 df = create.table_from_dataframe('##DFIdentityPK', pd.DataFrame({"ColumnA": [1,2]}), primary_key='sql') 

245 

246 #### create table using ColumnA as the primary key, after it was inferred to be the primary key 

247 df = create.table_from_dataframe('##DFInferPK', pd.DataFrame({"ColumnA": [1,2], "ColumnB": ["a","b"]}), primary_key='infer') 

248 """ 

249 

250 options = [None, "sql", "index", "infer"] 

251 if primary_key not in options: 

252 raise ValueError("primary_key must be one of: " + str(options)) 

253 

254 # determine primary key 

255 if primary_key is None: 

256 sql_primary_key = False 

257 primary_key_column = None 

258 elif primary_key == "sql": 

259 sql_primary_key = True 

260 primary_key_column = None 

261 elif primary_key == "index": 

262 sql_primary_key = False 

263 if not any(dataframe.index.names): 

264 dataframe.index.name = "_index" 

265 primary_key_column = list(dataframe.index.names) 

266 dataframe = dataframe.reset_index() 

267 elif primary_key == "infer": 

268 sql_primary_key = False 

269 primary_key_column = None 

270 

271 # infer SQL specifications from contents of dataframe 

272 dataframe, schema, not_nullable, pk = infer.sql(dataframe) 

273 _, dtypes = conversion.sql_spec(schema, dataframe) 

274 

275 # infer primary key column after best fit data types have been determined 

276 if primary_key == "infer": 

277 primary_key_column = pk 

278 

279 # add _time_insert column 

280 if self.include_metadata_timestamps: 

281 dtypes["_time_insert"] = "DATETIME2" 

282 

283 # create final SQL table 

284 self.table( 

285 table_name, 

286 dtypes, 

287 not_nullable=not_nullable, 

288 primary_key_column=primary_key_column, 

289 sql_primary_key=sql_primary_key, 

290 ) 

291 

292 # issue message for derived table 

293 pk_name = primary_key_column 

294 if sql_primary_key: 

295 pk_name = "_pk (SQL managed int identity column)" 

296 elif primary_key == "index": 

297 pk_name = str(primary_key_column) + " (dataframe index)" 

298 elif primary_key_column is not None: 

299 pk_name = primary_key_column + " (dataframe column)" 

300 else: 

301 pk_name = "None" 

302 msg = f""" 

303 Created table: {table_name} 

304 Primary key: {pk_name} 

305 Non-null columns: {not_nullable} 

306 Data types: {dtypes} 

307 """ 

308 warnings.warn(msg, custom_warnings.SQLObjectAdjustment) 

309 

310 # set primary key column as dataframe index 

311 if primary_key_column is not None: 

312 dataframe = dataframe.set_index(keys=primary_key_column) 

313 

314 # insert dataframe 

315 if insert_dataframe: 

316 cursor = self._connection.cursor() 

317 cursor.fast_executemany = True 

318 dataframe = conversion.insert_values( 

319 table_name, dataframe, self.include_metadata_timestamps, schema, cursor 

320 ) 

321 

322 return dataframe
